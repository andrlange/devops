New tables for AI settings, knowledge base, and chat history

Database Schema — New Tables (Flyway Migrations V3, V4, V5)
V3 — ai_settings
CREATE TABLE ai_settings (
id BIGINT PRIMARY KEY DEFAULT 1,
enabled BOOLEAN NOT NULL DEFAULT FALSE,
ollama_url VARCHAR(500),
model_name VARCHAR(200),
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
-- Singleton row: CHECK (id = 1)
-- Stores last used config, survives restarts
V4 — vet_knowledge (FTS)
CREATE TABLE vet_knowledge (
id BIGSERIAL PRIMARY KEY,
category VARCHAR(100) NOT NULL,
-- 'general_health', 'nutrition', 'vaccination',
-- 'emergency', 'post_visit', 'senior_care'
pet_type VARCHAR(50), -- NULL = all types
title VARCHAR(300) NOT NULL,
content TEXT NOT NULL,
search_vector TSVECTOR GENERATED ALWAYS AS (
to_tsvector('english', title || ' ' || content)
) STORED
);
CREATE INDEX idx_vet_knowledge_fts ON vet_knowledge USING GIN (search_vector);
CREATE INDEX idx_vet_knowledge_category ON vet_knowledge (category);
CREATE INDEX idx_vet_knowledge_pet_type ON vet_knowledge (pet_type);
FTS Query Example:
SELECT title, content, ts_rank(search_vector, query) AS rank
FROM vet_knowledge, to_tsquery('english', 'dog & limping & senior') query
WHERE search_vector @@ query
AND (pet_type IS NULL OR pet_type = 'Dog')
ORDER BY rank DESC LIMIT 3;
V5 — ai_chat_messages (Pet-persistent)
CREATE TABLE ai_chat_messages (
id BIGSERIAL PRIMARY KEY,
pet_id BIGINT NOT NULL REFERENCES pets(id),
session_id VARCHAR(100) NOT NULL,
role VARCHAR(20) NOT NULL, -- 'user' | 'assistant'
content TEXT NOT NULL,
knowledge_refs TEXT, -- JSON array of matched article titles
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX idx_chat_pet_id ON ai_chat_messages (pet_id);
CREATE INDEX idx_chat_session ON ai_chat_messages (session_id);
CREATE INDEX idx_chat_created ON ai_chat_messages (created_at DESC);

RAG Retrieval Flow

1 User asks: "What should I feed my senior cat?"
2 FTS query: to_tsquery('english', 'feed & senior & cat') WHERE pet_type IN (NULL, 'Cat')
3 Top 3 articles by ts_rank: "Senior Cat Nutrition", "Cat Diet Guide", "Aging Cat Care"
4 Build prompt: system prompt + article content + pet context (if pet chat) + user question
5 Stream response from Ollama via WebSocket to browser
